CONSTANT INV_SIZE = 100

DECLARE i,index:INTEGER
DECLARE inventory : ARRAY[1:INV_SIZE] OF inv
DECLARE in_stock  : ARRAY[1:INV_SIZE] OF stock

DECLARE cur_sz : INTEGER
cur_sz = INV_SIZE

TYPE inv
    DECLARE NAME: STRING
    DECLARE QUANTITY: INTEGER
ENDTYPE

TYPE stock
    DECLARE NAME: STRING
    DECLARE STCK: BOOLEAN
ENDTYPE

FUNCTION Sale(BYVAL Item : STRING, BYREF Quantity: INTEGER) RETURNS INTEGER

    DECLARE Flag:BOOLEAN
    Flag←FALSE
    FOR index←1 TO INV_SIZE
        IF inventory[index].NAME = Item THEN
            Flag←TRUE
            IF inventory[index].QUANTITY >= Quantity THEN
                inventory[index].QUANTITY←inventory[index].QUANTITY-Quantity
                RETURN 0
            ELSE
                RETURN 1
            ENDIF
        ENDIF
    NEXT index
    IF Flag=FALSE THEN
        RETURN 2
    ENDIF

ENDFUNCTION

FUNCTION AddItem(item : STRING, quan : INTEGER) RETURNS INTEGER
    IF cur_sz = INV_SIZE THEN
        RETURN 0
    ENDIF
    DECLARE i : INTEGER
    cur_sz ← cur_sz + 1
    DECLARE valid : BOOLEAN
    valid = FALSE
    i ← 1
    WHILE valid == FALSE AND i <= 100
        IF inventory[i].QUANTITY == -1 THEN
            valid ← TRUE
            inventory[i].NAME ← item
            inventory[i].QUANTITY ← quan
            in_stock[i].NAME ← item
            IF quan > 0 THEN
                in_stock[i].STCK ← FALSE
            ELSE
                in_stock[i].STCK ← TRUE
            ENDIF
        ENDIF
        i ← i + 1
    ENDWHILE
    IF valid = FALSE THEN
        RETURN 0
    ELSE 
        RETURN 1
    ENDIF
ENDFUNCTION
FUNCTION RemoveItem(item : STRING) RETURNS INTEGER
    DECLARE valid : BOOLEAN
    valid ← FALSE
    DECLARE i : INTEGER
    i ← 1
    IF cur_sz = 0 THEN
        RETURN 0
    ENDIF
    WHILE i <= 100 AND valid = FALSE
        IF inventory[i].NAME = item THEN
            inventory[i].NAME = "NaN"
            inventory[i].QUANTITY = -1
            in_stock[i].NAME = "NaN"
            in_stock[i].STCK = FALSE
            valid = TRUE
        ENDIF
        i ← i + 1
    ENDWHILE
    IF valid = FALSE THEN
        RETURN 0
    ELSE
        RETURN 1
    ENDIF
ENDFUNCTION

PROCEDURE InitialiseInventory()
    DECLARE i : INTEGER
    FOR i ← 1 TO INV_SIZE
        inventory[i].NAME ← "Item" & i
        inventory[i].QUANTITY ← INT(RAND(500))
        in_stock[i].NAME ← inventory[i].NAME
        IF inventory[i].QUANTITY > 0 THEN
            in_stock[i].STCK ← TRUE
        ELSE
            in_stock[i].STCK ← FALSE
        ENDIF
    NEXT i
ENDPROCEDURE

PROCEDURE main()
    CALL InitialiseInventory()

    DECLARE qnt,retrn:INTEGER
    DECLARE item:STRING
    item←"Item1"
    qnt←10
    retrn←Sale(item,qnt)
    IF retrn = 0 THEN
        OUTPUT "Successfull"
    ENDIF
    IF retrn = 1 THEN
        OUTPUT "Not enough in stock of"&item
    ENDIF
    IF retrn = 2 THEN
        OUTPUT item & "is not present in stock"
    ENDIF
    
ENDPROCEDURE
