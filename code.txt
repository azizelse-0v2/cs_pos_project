CONSTANT INV_SIZE = 100

DECLARE i,index,sales_index,cur_sz:INTEGER
DECLARE inventory : ARRAY[1:INV_SIZE] OF inv
DECLARE in_stock  : ARRAY[1:INV_SIZE] OF stock//for 2d array
DECLARE sales     : ARRAY[1:INV_SIZE] OF inv
sales_index←1
cur_sz ← INV_SIZE

TYPE inv
    DECLARE NAME: STRING
    DECLARE QUANTITY: INTEGER
ENDTYPE

TYPE stock
    DECLARE NAME: STRING
    DECLARE STCK: BOOLEAN
ENDTYPE

FUNCTION Sale(BYVAL Item : STRING, BYREF Quantity: INTEGER) RETURNS INTEGER

    DECLARE Flag:BOOLEAN
    Flag←FALSE
    FOR index←1 TO INV_SIZE
        IF inventory[index].NAME = Item THEN
            Flag←TRUE
            IF inventory[index].QUANTITY >= Quantity THEN
                inventory[index].QUANTITY←inventory[index].QUANTITY-Quantity
                sales[sales_index].NAME ← Item
                sales[sales_index].QUANTITY ← Quantity
                sales_index←sales_index+1
                RETURN 0
            ELSE
                RETURN 1
            ENDIF
        ENDIF
    NEXT index
    IF Flag=FALSE THEN
        RETURN 2
    ENDIF

ENDFUNCTION

FUNCTION AddItem(item : STRING, quan : INTEGER) RETURNS INTEGER
    IF cur_sz = INV_SIZE THEN
        RETURN 1
    ENDIF
    DECLARE i : INTEGER
    cur_sz ← cur_sz + 1
    DECLARE valid : BOOLEAN
    valid ← FALSE
    i ← 1
    WHILE valid = FALSE AND i <= INV_SIZE
        IF inventory[i].QUANTITY = -1 THEN
            valid ← TRUE
            inventory[i].NAME ← item
            inventory[i].QUANTITY ← quan
            in_stock[i].NAME ← item
            IF quan > 0 THEN
                in_stock[i].STCK ← FALSE
            ELSE
                in_stock[i].STCK ← TRUE
            ENDIF
        ENDIF
        i ← i + 1
    ENDWHILE
    IF valid = FALSE THEN
        RETURN 2
    ELSE 
        RETURN 0
    ENDIF
ENDFUNCTION

FUNCTION RemoveItem(item : STRING) RETURNS INTEGER
    DECLARE valid : BOOLEAN
    valid ← FALSE
    DECLARE i : INTEGER
    i ← 1
    IF cur_sz = 0 THEN
        RETURN 1
    ENDIF
    WHILE i <= 100 AND valid = FALSE
        IF inventory[i].NAME = item THEN
            cur_sz ← cur_sz - 1
            inventory[i].NAME ← "NaN"
            inventory[i].QUANTITY ← -1
            in_stock[i].NAME ← "NaN"
            in_stock[i].STCK ← FALSE
            valid ← TRUE
        ENDIF
        i ← i + 1
    ENDWHILE
    IF valid = FALSE THEN
        RETURN 1
    ELSE
        RETURN 0
    ENDIF
ENDFUNCTION

PROCEDURE SalesReport()
    FOR index←1 TO sales_index-1
        OUTPUT NUM_TO_STR(index) & " Sold " & NUM_TO_STR(sales[index].QUANTITY) & " many of " & sales[index].NAME
    NEXT index
ENDPROCEDURE

PROCEDURE InventoryReport()
    FOR index←1 TO INV_SIZE
        CASE OF inventory[index].NAME
            "NaN": OUTPUT ""
            OTHERWISE: OUTPUT "There are " & NUM_TO_STR(inventory[index].QUANTITY) & " of " & inventory[index].NAME
        ENDCASE
    NEXT index
ENDPROCEDURE

//vibecoded function
PROCEDURE InitialiseInventory()
    DECLARE i : INTEGER
    FOR i ← 1 TO INV_SIZE
        inventory[i].NAME ← "Item" & NUM_TO_STR(i)
        inventory[i].QUANTITY ← INT(RAND(500))
        in_stock[i].NAME ← inventory[i].NAME
        IF inventory[i].QUANTITY > 0 THEN
            in_stock[i].STCK ← TRUE
        ELSE
            in_stock[i].STCK ← FALSE
        ENDIF
    NEXT i
ENDPROCEDURE

PROCEDURE Handler(State : INTEGER)
    DECLARE item: STRING
    DECLARE quantity,res: INTEGER
    IF State <= 3 THEN
        OUTPUT "Item: "
        INPUT item
        IF State <= 2 THEN
            OUTPUT "Quantity: "
            INPUT quantity

            IF State = 1 THEN
                res←Sale(item,quantity)
                IF res = 0 THEN
                    OUTPUT "SUCCESSFUL"
                ENDIF
                IF res = 1 THEN
                    OUTPUT "The stock has less quantity than needed"
                ENDIF
                IF res = 2 THEN
                    OUTPUT "The item not found in the stock"
                ENDIF
            ENDIF


            IF State = 2 THEN
                res←AddItem(item, quantity)
                IF res=0 THEN
                    OUTPUT "SUCCESSFUL"
                ENDIF
                IF res = 1 THEN
                    OUTPUT "The array is full"
                ENDIF
                IF res = 2 THEN
                    OUTPUT "No undeclared items"
                ENDIF
            ENDIF
        ELSE
            res←RemoveItem(item)
            IF res = 0 THEN
                OUTPUT "SUCCESSFUL"
            ENDIF
            IF res = 1 THEN
                OUTPUT "The item isn't in the array"
            ENDIF
        ENDIF
    ELSE
        IF State = 4 THEN
            CALL SalesReport()
        ENDIF
        IF State = 5 THEN
            CALL InventoryReport()
        ENDIF
    ENDIF

ENDPROCEDURE

PROCEDURE menu()
    DECLARE inp: INTEGER
    OUTPUT "Choose of the following:\n1:Sale\n2:AddItem\n3:RemoveItem\n4:Salesreport\n5:InventoryReport"
    INPUT inp
    CASE OF inp
        1 : CALL Handler(1)
        2 : CALL Handler(2)
        3 : CALL Handler(3)
        4 : CALL Handler(4)
        5 : CALL Handler(5)
        OTHERWISE: OUTPUT "Error" CALL menu()
    ENDCASE
ENDPROCEDURE

PROCEDURE main()
    CALL menu()
    CALL main()
ENDPROCEDURE

CALL InitialiseInventory()
CALL main()
